/**
 * GET Webhook API: Digunakan oleh aplikasi React untuk mengambil seluruh data
 * dari Google Sheet database ke web UI secara instan.
 */
function doGet(e) {
  try {
    const ss = getSpreadsheet();
    if (!ss) {
      return createJsonResponse({ success: false, message: "Database tidak ditemukan. Silakan jalankan setupDatabase() terlebih dahulu." });
    }

    const data = {
      users: getSheetDataAsJson(ss, "USER_ROLE"),
      overtime: getSheetDataAsJson(ss, "OVERTIME"),
      karyawan: getSheetDataAsJson(ss, "DATA_KARYAWAN"),
      auditLogs: getSheetDataAsJson(ss, "AUDIT_LOG")
    };

    return createJsonResponse({ success: true, data: data });
  } catch (error) {
    return createJsonResponse({ success: false, error: error.toString() });
  }
}

/**
 * POST Webhook API: Menerima pengajuan lembur baru, persetujuan manager,
 * revisi, edit database master, dan log audit dari web app.
 */
function doPost(e) {
  try {
    const ss = getSpreadsheet();
    if (!ss) {
      return createJsonResponse({ success: false, message: "Spreadsheet database belum siap." });
    }

    const postData = JSON.parse(e.postData.contents);
    const action = postData.action; // jenis aksi: 'CREATE_OT', 'UPDATE_STATUS', 'SAVE_KARYAWAN', 'SAVE_USER'
    const payload = postData.payload;

    if (action === "CREATE_OT") {
      const sheet = ss.getSheetByName("OVERTIME");
      // Payload berisi array of records (multi-employee)
      payload.forEach(item => {
        sheet.appendRow([
          item.idDoc, item.tanggalPengajuan, item.tanggalLembur, item.areaLembur, item.divisi,
          item.jenisLembur, item.namaKaryawan, item.idKaryawan, item.jabatan, item.jamMulai,
          item.jamSelesai, item.durasiLembur, item.nominal, item.statusDocument, item.alasanLembur, item.catatanManager
        ]);
      });
      writeLogToSheet(ss, postData.currentUser, "pembuatan berkas", "Menambahkan pengajuan dokumen lembur " + payload[0].idDoc);

    } else if (action === "UPDATE_STATUS") {
      const sheet = ss.getSheetByName("OVERTIME");
      const rows = sheet.getDataRange().getValues();
      const docId = payload.idDoc;
      const newStatus = payload.statusDocument;
      const comment = payload.catatanManager || "";

      // Cari baris yang cocok dengan docId lalu ubah status & catatan manager
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === docId) {
          sheet.getRange(i + 1, 14).setValue(newStatus); // Kolom N (status document)
          sheet.getRange(i + 1, 16).setValue(comment);   // Kolom P (catatan manager)
        }
      }
      writeLogToSheet(ss, postData.currentUser, "mutasi status persetujuan", "Mengubah status " + docId + " menjadi " + newStatus.toUpperCase());

    } else if (action === "SAVE_KARYAWAN") {
      const sheet = ss.getSheetByName("DATA_KARYAWAN");
      const rows = sheet.getDataRange().getValues();
      const idKaryawan = payload.idKaryawan;
      let foundIndex = -1;

      for (let i = 1; i < rows.length; i++) {
        if (rows[i][1] === idKaryawan) {
          foundIndex = i + 1;
          break;
        }
      }

      const rowValues = [payload.nik, payload.idKaryawan, payload.namaKaryawan, payload.divisi, payload.areaKerja, payload.jabatan];
      if (foundIndex !== -1) {
        sheet.getRange(foundIndex, 1, 1, rowValues.length).setValues([rowValues]);
        writeLogToSheet(ss, postData.currentUser, "revisi", "Mengubah data karyawan " + payload.namaKaryawan);
      } else {
        sheet.appendRow(rowValues);
        writeLogToSheet(ss, postData.currentUser, "pembuatan berkas", "Mendaftarkan karyawan baru " + payload.namaKaryawan);
      }
    }

    return createJsonResponse({ success: true, message: "Aksi " + action + " sukses disimpan ke Google Sheets." });
  } catch (error) {
    return createJsonResponse({ success: false, error: error.toString() });
  }
}

/**
 * Mengambil referensi Spreadsheet dari folder target
 */
function getSpreadsheet() {
  const folder = DriveApp.getFolderById(TARGET_FOLDER_ID);
  const files = folder.getFilesByName(SPREADSHEET_NAME);
  if (files.hasNext()) {
    return SpreadsheetApp.openById(files.next().getId());
  }
  return null;
}

/**
 * Membaca data sheet dan mengkonversinya menjadi Array of JSON Objects
 */
function getSheetDataAsJson(ss, sheetName) {
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];
  
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];

  const headers = data[0];
  const jsonResult = [];

  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const obj = {};
    for (let j = 0; j < headers.length; j++) {
      // Ubah nama header ke format CamelCase/SnakeCase yang sesuai jika diperlukan
      let key = headers[j].toString().trim();
      // Khusus untuk nama kolom dengan spasi seperti "id doc" diubah ke "idDoc"
      if (key === "id doc") key = "idDoc";
      else if (key === "tanggal pengajuan") key = "tanggalPengajuan";
      else if (key === "tanggal lembur") key = "tanggalLembur";
      else if (key === "area lembur") key = "areaLembur";
      else if (key === "jenis lembur") key = "jenisLembur";
      else if (key === "nama karyawan") key = "namaKaryawan";
      else if (key === "id karyawan") key = "idKaryawan";
      else if (key === "jam mulai") key = "jamMulai";
      else if (key === "jam selesai") key = "jamSelesai";
      else if (key === "durasi lembur") key = "durasiLembur";
      else if (key === "status document") key = "statusDocument";
      else if (key === "alasan lembur") key = "alasanLembur";
      else if (key === "catatan manager") key = "catatanManager";
      
      obj[key] = row[j];
    }
    jsonResult.push(obj);
  }
  return jsonResult;
}

/**
 * Menulis log kejadian ke tab sheet AUDIT_LOG secara otomatis
 */
function writeLogToSheet(ss, user, action, details) {
  const sheet = ss.getSheetByName("AUDIT_LOG");
  const timestamp = new Date().toISOString().replace("T", " ").substring(0, 19);
  const userStr = user ? user.username : "system";
  sheet.appendRow([timestamp, userStr, action, details]);
}

/**
 * Helper pembentuk response JSON aman untuk Web App CORS
 */
function createJsonResponse(output) {
  return ContentService.createTextOutput(JSON.stringify(output))
    .setMimeType(ContentService.MimeType.JSON);
}
